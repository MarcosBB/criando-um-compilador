function void mdc(int n, int m, int r) {
    if (n == 0) {
        r << m;
    } else if (m == 0) {
        r << n;
    } else if (m > n) {
        mdc(n, m % n, r);
    } else {
        mdc(m, n % m, r);
    }
}

function void main() {
    int n << 48;
    int m << 18;
    int r;
    // mdc(48, 18, r);
    mdc(n, m, r);
}
